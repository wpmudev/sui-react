export const ToastButton = () => {
	const notification = useNotifications()
	return (
		<Button
			type="secondary"
			colorScheme="blue"
			isSmall={true}
			onClick={() => {
				notification.push({
					title: "Toast Title",
					message: "This is notification message.",
					icon: "CheckAlt",
				})
			}}
		>
			Show Notification
		</Button>
	)
}

export const TimeOutToastButton = () => {
	const notification = useNotifications()
	return (
		<Button
			type="secondary"
			colorScheme="blue"
			isSmall={true}
			onClick={() => {
				notification.push({
					title: "Toast Title",
					message: "This is notification message.",
					icon: "CheckAlt",
					timeout: 1000,
				})
			}}
		>
			Show Notification
		</Button>
	)
}
