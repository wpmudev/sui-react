export const phpSampleCode = `function recent_posts_shortcode() {
    $query = new WP_Query(array(
        'post_type' => 'post',
        'posts_per_page' => 5,
        'orderby' => 'date',
        'order' => 'DESC',
    ));

    $output = '<ul>';
    while ($query->have_posts()) {
        $query->the_post();
        $title = get_the_title();
        $link = get_permalink();
        $date = get_the_date();
        $output .= "<li><a href='$link'>$title</a> - $date</li>";
    }
    $output .= '</ul>';
    wp_reset_postdata();

    return $output;

}

add_shortcode('recent_posts', 'recent_posts_shortcode'); `
