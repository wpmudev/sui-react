		</p>

	<PropSection title="customWidth" type="number">

			Custom width for the input in <Code isSmall={true}>pixels</Code>
		</p>
	</PropSection>

	<PropSection title="type" type="string" defaultValue="text">
		<p>

	</PropSection>

	<PropSection title="defaultValue" type="string">
		<p>
			The initial value of the input.
		</p>
	</PropSection>

	<PropSection title="hint" type="string">
		<p>
			A text to appear initially in the input to give the user a hint ( a part of the text that he should type )
		</p>
	</PropSection>

	<PropSection title="iconHint" type="string">
		<p>
			A hint that specifies the icon purpose, it appears in a tooltip
		</p>
	</PropSection>

	<PropSection title="iconTooltipWidth" type="number">
		<p>
			A custom width for the icon Tooltip hint
		</p>
	</PropSection>

	<PropSection title="isMultiline" type="boolean" defaultValue="false">

			If set to <Code isSmall={true}>true</Code>, it will use <Code isSmall={true}>textarea</Code> tag instead of the <Code isSmall={true}>input</Code>

	</PropSection>
	
	<PropSection title="isReadOnly" type="boolean" defaultValue="false">

			If set to <Code isSmall={true}>true</Code> the input text won't be editable

	<PropSection title="isRequired" type="boolean" defaultValue="false">
		<p>
			Whether the input is required or not
		</p>
	</PropSection>

	<PropSection title="onChange" type="function">
		<p>A callback function to be invoked when the input text changes, the first parameter is the change event object</p>
	</PropSection>

	<PropSection title="onClick" type="function">
		<p>
			A callback function to be invoked when the input is clicked
		</p>
	</PropSection>

	<PropSection title="onClickIcon" type="function">
		<p>
			A callback function to be invoked when the input icon is clicked
		</p>
	</PropSection>

	<PropSection title="allowClear" type="boolean" defaultValue="false">
		<p>Whether to display the clear button or not</p>
	</PropSection>

	<PropSection title="onClear" type="function">
		<p>A callback function to be invoked when the clear button is clicked</p>
	</PropSection>
	
	<PropSection title="icon" type="string">
		<p>An icon to be displayed in the input</p>
	</PropSection>

	<PropSection title="iconPosition" type="string" defaultValue="start">

			<span label="start"/>
			<span label="end"/>
		</List>
	</PropSection>
	
	<PropSection title="isError" type="boolean" defaultValue="false">
		<p>Reflects the error status in the field.</p>
	</PropSection>
	
	<PropSection title="isSmall" type="boolean" defaultValue="false">
		<p>Set if the input is small.</p>
	</PropSection>
	
	<PropSection title="isDisabled" type="boolean" defaultValue="false">

	<PropSection title="disableInteractions" type="boolean" defaultValue="false">

	<PropSection title="className" type="string">

	</PropSection>

	<PropSection title="inputClass" type="string">

		</p>
	</PropSection>

		</p>
	</PropSection>

	<PropSection title="pattern" type="string">

	<PropSection title="validateOnMount" type="string" defaultValue="false">

	<PropSection title="onFocus" type="function">
